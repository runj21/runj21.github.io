                class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        left, right = 0, len(nums) - 1
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] == target:
                return mid
            elif nums[mid] < target:
                left = mid + 1
            else:
                right = mid - 1
        return left

                复杂度分析： 时间 O(log n)，空间 O(1)

                💭 我的感想： 二分法的主要思路是对一个连续单调的数组进行分割，最为关键的一点是区间的选择，一般采用两端封闭区间，即 [a,b] ，对应得判断条件就是 left <= right 。当中间值大于 target 时，区间选左边，right = mid -1，反之 left = mid + 1 。 对于这道题而言，若二分法找到 target 直接输出 mid 。反之，当循环结束时，存在 left = right + 1 。查找过程中，左侧 [0,left-1] 的值都小于 target， 右侧 [right+1,len(nums)-1] 的值都大于 target。因此 target 取 left 或者 right+1 。

                class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        m, n = len(matrix), len(matrix[0])
        left, right = 0, m * n - 1
        while left <= right:
            mid = (left + right) // 2
            val = matrix[mid // n][mid % n]
            if val == target:
                return True
            elif val < target:
                left = mid + 1
            else:
                right = mid - 1
        return False

                复杂度分析： 时间 O(log(m×n))，空间 O(1)

                class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        res = [-1,-1]
        left, right = 0, len(nums)-1
        while left<=right:
            mid = (left+right)//2
            if nums[mid] == target:
                p1 = mid
                p2 = mid
                while p1>=0 and nums[p1] == target:
                    p1 += -1
                while p2<=(len(nums)-1) and nums[p2] == target:
                    p2 += 1
                return [p1+1,p2-1]
            elif nums[mid]>target:
                right = mid - 1
            else:
                left = mid + 1
        return res

                复杂度分析： 时间 O(n)，空间 O(1)

                💭 我的感想： 二分法，先找到一个目标值的位置，再使用 while 向前、向后搜寻 target 的范围。

                class Solution:
    def search(self, nums: List[int], target: int) -> int:
        left, right = 0, len(nums) - 1
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] == target:
                return mid
            if nums[left] <= nums[mid]: 
                if nums[left] <= target < nums[mid]:
                    right = mid - 1
                else:
                    left = mid + 1
            else: 
                if nums[mid] < target <= nums[right]:

                    right = mid - 1

                class Solution:
    def findMin(self, nums: List[int]) -> int:

                class Solution:
    def findMedianSortedArrays(self, nums1: List[int], nums2: List[int]) -> float:
        if len(nums1) > len(nums2):
            nums1, nums2 = nums2, nums1
        m, n = len(nums1), len(nums2)
        left, right = 0, m
        while left <= right:
            i = (left + right) // 2
            j = (m + n + 1) // 2 - i
            nums1_left = nums1[i - 1] if i > 0 else float('-inf')
            nums1_right = nums1[i] if i < m else float('inf')
            nums2_left = nums2[j - 1] if j > 0 else float('-inf')
            nums2_right = nums2[j] if j < n else float('inf')
            if nums1_left <= nums2_right and nums2_left <= nums1_right:
                if (m + n) % 2 == 0:
                    return (max(nums1_left, nums2_left) + min(nums1_right, nums2_right)) / 2
                else:
                    return max(nums1_left, nums2_left)
            elif nums1_left > nums2_right:
                right = i - 1
            else:
                left = i + 1

                复杂度分析： 时间 O(log(min(m, n)))，空间 O(1)

                💭 我的感想： 寻找两个正序数组的中位数，二分查找。我们希望在两个有序数组 nums1 和 nums2 中找到一条分割线，使得：分割线左边的元素个数 等于（或比右边多 1，当总长度为奇数时）右边的元素个数；并且左边所有元素 都 ≤ 右边所有元素；一旦满足这两个条件，中位数就可以由分割线两侧的四个关键值直接计算出来。难度较大。

                class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        slow = 0
        for fast in range(len(nums)):
            if nums[fast] != 0:
                nums[slow], nums[fast] = nums[fast], nums[slow]

                class Solution:
    def maxArea(self, height: List[int]) -> int:
        left, right = 0, len(height) - 1
        max_area = 0
        while left < right:
            area = min(height[left], height[right]) * (right - left)
            max_area = max(max_area, area)
            if height[left] < height[right]:
                left += 1
            else:
                right -= 1
        return max_area

                复杂度分析： 时间 O(n)，空间 O(1)

                class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        n = len(nums)
        res = []
        for i in range(n - 2):
            if nums[i] > 0:
                break
            if i > 0 and nums[i] == nums[i-1]:
                continue
            left, right = i + 1, n - 1
            while left < right:
                total = nums[i] + nums[left] + nums[right]
                if total == 0:
                    res.append([nums[i], nums[left], nums[right]])
                    while left < right and nums[left] == nums[left+1]:
                        left += 1
                    while left < right and nums[right] == nums[right-1]:
                        right -= 1
                    left += 1
                    right -= 1
                elif total < 0:
                    left += 1
                else:
                    right -= 1
        return res

                复杂度分析： 时间 O(n²)，空间 O(log n)（排序所需栈空间）

                💭 我的感想： 三数之和，双指针，关键在于去重。首先，对整个数组进行排序，然后固定一个 i ，对其余两个使用双指针 left right 。若 nums[i] == nums[i-1] 说明之前已经计算过这种情况(当时，left = i 且 i = i-1 ……)跳过 nums[i] 这种情况。nums[right] == nums[right-1] 与 nums[left] == nums[left+1] 也是同理。

                class Solution:
    def trap(self, height: List[int]) -> int:
        left, right = 0, len(height) - 1
        left_max = right_max = 0
        water = 0
        while left < right:
            if height[left] < height[right]:
                if height[left] >= left_max:
                    left_max = height[left]
                else:
                    water += left_max - height[left]
                left += 1
            else:
                if height[right] >= right_max:
                    right_max = height[right]
                else:
                    water += right_max - height[right]
                right -= 1
        return water

                复杂度分析： 时间 O(n)，空间 O(1)

                class Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]:
        from collections import Counter
        len_p, len_s = len(p), len(s)
        if len_p > len_s:
            return []
        p_count = Counter(p)
        s_count = Counter(s[:len_p])
        res = []
        if s_count == p_count:
            res.append(0)
        for i in range(len_p, len_s):
            s_count[s[i]] += 1
            left_char = s[i - len_p]
            s_count[left_char] -= 1
            if s_count[left_char] == 0:
                del s_count[left_char]
            if s_count == p_count:
                res.append(i - len_p + 1)
        return res

                class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        char_set = set()
        left = 0
        max_len = 0
        for right in range(len(s)):
            while s[right] in char_set:
                char_set.remove(s[left])
                left += 1
            char_set.add(s[right])

    def setZeroes(self, matrix: List[List[int]]) -> None:

        first_row_zero = any(matrix[0][j] == 0 for j in range(n))

                class Solution:
    def sortedArrayToBST(self, nums: List[int]) -> Optional[TreeNode]:
        def build(left, right):
            if left > right:
                return None
            mid = (left + right) // 2
            root = TreeNode(nums[mid])
            root.left = build(left, mid - 1)
            root.right = build(mid + 1, right)
            return root
        return build(0, len(nums)-1)

                class Solution:
    def climbStairs(self, n: int) -> int:

        a, b = 1, 2
        for i in range(2, n):

                118. 杨辉三角#

                题目描述： 给定一个非负整数 numRows，生成杨辉三角的前 numRows 行。在杨辉三角中，每个数是它左上方和右上方的数的和。

                class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        triangle = []
        for i in range(numRows):
            row = [1] * (i + 1)
            for j in range(1, i):
                row[j] = triangle[i - 1][j - 1] + triangle[i - 1][j]
            triangle.append(row)
        return triangle

                198. 打家劫舍#

                题目描述： 你是一个专业的小偷，计划偷窃沿街的房屋。每间房内都藏有一定的现金，影响你偷窃的唯一制约因素就是相邻的房屋装有相互连通的防盗系统——如果两间相邻的房屋在同一晚上被小偷闯入，系统会自动报警。给定一个代表每个房屋存放金额的数组，计算你在不触动警报装置的情况下能够偷窃到的最高金额。

                解题思路： 线性 DP。dp[i] = max(dp[i-1], dp[i-2] + nums[i])，即第 i 间房的最大收益 = max(不偷第 i 间, 偷第 i 间)。可以用两个变量滚动优化。

                class Solution:
    def rob(self, nums: List[int]) -> int:
        n = len(nums)
        if n == 1:
            return nums[0]
        dp = [0] * n
        dp[0] = nums[0]
        dp[1] = max(nums[0], nums[1])
        for i in range(2, n):
            dp[i] = max(dp[i - 1], dp[i - 2] + nums[i])
        return dp[n - 1]

                279. 完全平方数#

                题目描述： 给定正整数 n，找到若干个完全平方数（如 1, 4, 9, 16, …）使得它们的和等于 n，返回组成和的最少完全平方数个数。

                解题思路： 完全背包 DP。dp[i] 表示组成 i 的最少完全平方数个数。对每个 i，枚举所有 j*j <= i，取 dp[i - j*j] + 1 的最小值。

                class Solution:
    def numSquares(self, n: int) -> int:
        dp = [float('inf')] * (n + 1)
        dp[0] = 0
        for i in range(1, n + 1):
            j = 1
            while j * j <= i:
                dp[i] = min(dp[i], dp[i - j * j] + 1)
                j += 1
        return dp[n]
